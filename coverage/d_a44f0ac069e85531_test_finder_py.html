1"""Tests for the `finder` module.""" 

2 

3import pytest 

4 

5from griffe.finder import ModuleFinder 

6from tests.helpers import temporary_pypackage 

7 

8 

9@pytest.mark.parametrize( 

10 ("pypackage", "module", "add_to_search_path", "expected_top_name", "expected_top_path"), 

11 [ 

12 (("a", ["b.py"]), "a/b.py", True, "a", "a/__init__.py"), 

13 (("a", ["b.py"]), "a/b.py", False, "a", "a/__init__.py"), 

14 (("a/b", ["c.py"]), "a/b/c.py", True, "a", "a"), 

15 (("a/b", ["c.py"]), "a/b/c.py", False, "b", "a/b/__init__.py"), 

16 ], 

17) 

18def test_find_module_with_path(pypackage, module, add_to_search_path, expected_top_name, expected_top_path): 

19 """Check that the finder can find modules using strings and Paths. 

20 

21 Parameters: 

22 pypackage: A temporary package (metadata) on the file system (parametrized). 

23 module: The module path to load (parametrized). 

24 add_to_search_path: Whether to add the temporary package parent path to the finder search paths (parametrized). 

25 expected_top_name: Expected top module name (parametrized). 

26 expected_top_path: Expected top module path (parametrized). 

27 """ 

28 with temporary_pypackage(*pypackage) as tmp_package: 

29 finder = ModuleFinder(search_paths=[tmp_package.tmpdir] if add_to_search_path else None) 

30 _, package = finder.find_spec(tmp_package.tmpdir / module) 

31 assert package.name == expected_top_name 

32 if package.is_namespace: 

33 assert package.path == [tmp_package.tmpdir / expected_top_path] 

34 else: 

35 assert package.path == tmp_package.tmpdir / expected_top_path
